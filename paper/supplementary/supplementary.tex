\subsection*{Supplementary Table S2: ArchaicPainter HMM parameter values}

Default parameter values used in all analyses, with coalescent-theoretic justifications.

\begin{table}[h]
\centering
\begin{tabular}{lll}
\hline
\textbf{Parameter} & \textbf{Value} & \textbf{Justification} \\
\hline
$t_\mathrm{admixture}$ & 1,724 generations & $\sim$50 kya / 29 yr generation time \\
$t_\mathrm{split}$ & 19,000 generations & $\sim$550 kya / 29 yr generation time \\
Match-length ratio & $11\times$ & $t_\mathrm{split} / t_\mathrm{admixture}$ \\
Expected NEA segment & 58 kb & $1 / t_\mathrm{admixture}$ at 1 cM/Mb \\
$\pi_\mathrm{NEA}$ prior & 0.02 & Published 2\% Neanderthal ancestry fraction \\
$\theta_\mathrm{emit}$ & 0.01 & Per-site mismatch probability between query and archaic reference (captures sequencing error, ancient DNA damage, and reference--donor divergence; not the per-generation germline mutation rate $\mu \approx 10^{-8}$) \\
Merge gap & 10 kb & Post-Viterbi gap filling (calibrated on simulation) \\
Min segment & 10 kb & Post-Viterbi minimum segment filter \\
\hline
\end{tabular}
\caption{ArchaicPainter HMM default parameters. Generation time: 29 years.
Admixture time: $\sim$50 kya (Prüfer et al.\ 2014). AMH--archaic split: $\sim$550 kya.}
\label{tab:params}
\end{table}

%% ── Supplementary Table S3 ──────────────────────────────────────────────────
\subsection*{Supplementary Table S3: Parameter sensitivity analysis}

F1 scores under $\pm$20\% perturbation of key HMM parameters, evaluated on the
50-replicate simulation benchmark (seeds 42--91).

\begin{table}[h]
\centering
\begin{tabular}{lcc}
\hline
\textbf{Configuration} & \textbf{Mean F1} & \textbf{Std F1} \\
\hline
Standard ArchaicPainter (default) & 0.480 & 0.095 \\
CEU YRI reference (vs.\ standard YRI) & 0.507 & 0.092 \\
$\theta_\mathrm{emit} \times 2$ (high, $\theta=0.02$) & 0.470 & 0.100 \\
No segment merging & 0.488 & 0.095 \\
Positive-only emission & 0.154 & 0.079 \\
\hline
\end{tabular}
\caption{Parameter sensitivity: all perturbations except positive-only emission
yield F1 differences $<0.04$ from the default configuration, confirming robustness
to parameter choices within the biologically motivated range.
The positive-only model achieves lower simulation F1 (0.154 vs.\ 0.480) because
mismatch signals carry essential discriminative information when the reference
matches the true introgression source (see main text for domain discussion).}
\label{tab:sensitivity}
\end{table}

\subsection*{Supplementary Figure S1: ArchaicPainter HMM architecture and Viterbi decoding}

\begin{figure}[!ht]
\centering
\includegraphics[width=\textwidth]{figures/sfig1_hmm_viterbi}
\caption{\textbf{ArchaicPainter HMM architecture and example Viterbi decoding.}
\textbf{(a)} Three-state HMM with states AMH (blue), NEA (orange), and DEN (green).
Transition probabilities follow $\exp(-d \cdot t)$ where $d$ is the recombination distance
in Morgans and $t$ is the admixture time in generations.
Emission models are shown for both the standard bidirectional mode (simulation, where
the reference matches the true donor) and the positive-only log-score mode (real data, where
reference--donor divergence requires down-weighting mismatch evidence; see main text).
\textbf{(b)} Default HMM parameter values with biological justifications.
Core HMM parameters ($t_\mathrm{admix}$, $t_\mathrm{split}$, $\pi_\mathrm{NEA}$, $\theta_\mathrm{emit}$) are derived from published coalescent theory;
post-processing thresholds (merge gap, min segment) are calibrated
by grid search on simulation to maximise F1.
\textbf{(c)} Example Viterbi decoding output on a single simulated haplotype (seed 42,
5~Mb region, 2\% Neanderthal admixture).
The true introgressed segment (SNPs 60--140, orange shading) generates a dense cluster
of archaic reference matches (top track, orange bars), which translates into elevated
NEA posterior probability (middle track, red curve), and is correctly captured by the
Viterbi path (bottom track, black step function, SNPs 55--145).
The slight over-extension of the Viterbi path relative to the true segment reflects
the 10-kb gap-merging post-processing step.}
\label{sfig:hmm}
\end{figure}

\subsection*{Supplementary Figure S2: Three-state Neanderthal/Denisovan source attribution}

\begin{figure}[!ht]
\centering
\includegraphics[width=\textwidth]{figures/sfig2_attribution}
\caption{\textbf{Three-state ArchaicPainter source attribution results on two-source
simulation (20 replicates, seeds 100--119).}
Simulation demography: 5~Mb region, 2\% Neanderthal (NEA) + 1\% Denisovan (DEN)
simultaneous admixture, matched simulated archaic reference panels.
\textbf{Top row (orange)}: NEA segment-level F1 score (mean $0.260 \pm 0.109$),
NEA source attribution accuracy (mean $26.1\% \pm 13.5\%$), and NEA-to-DEN
cross-contamination rate (fraction of NEA predictions overlapping true DEN regions;
mean $11.9\%$).
\textbf{Bottom row (green)}: DEN F1 score (mean $0.188 \pm 0.105$),
DEN attribution accuracy (mean $15.2\% \pm 10.1\%$), and DEN-to-NEA
cross-contamination (mean $19.3\%$).
Attribution accuracies above chance ($>$10\% for both sources) confirm that the
three-state HMM extracts lineage-discriminating information from the available
archaic-lineage-specific alleles, despite the limited NEA--DEN divergence
($\sim$700 kya; Prüfer et al.\ 2014).
Box plots show median and IQR; points show individual replicates; dashed red line
indicates mean.}
\label{sfig:attribution}
\end{figure}
